SELECT kreatura.nazwa, DATEDIFF(wyprawa.data_rozpoczecia, kreatura.dataUr) AS wiek_w_dniach FROM kreatura RIGHT JOIN uczestnicy ON kreatura.idKreatury = uczestnicy.id_uczestnika RIGHT JOIN wyprawa ON uczestnicy.id_wyprawy = wyprawa.id_wyprawy RIGHT JOIN etapy_wyprawy ON wyprawa.id_wyprawy = etapy_wyprawy.idWyprawy RIGHT JOIN sektor ON etapy_wyprawy.sektor = sektor.id_sektora WHERE sektor.id_sektora = 6 GROUP BY kreatura.nazwa;
